Second Highest Salary

Oct 17, 2017 1 min read SQL, LeetCode 
CREATE DATABASE IF NOT EXISTS `LeetCode`;
USE `LeetCode`;
DROP TABLE IF EXISTS `Employee`;
CREATE TABLE IF NOT EXISTS `Employee` (
  `Id` INT, 
  `Salary` INT
);
INSERT INTO `Employee`
VALUES
  (1, 100), 
  (2, 200), 
  (3, 300);
DROP FUNCTION IF EXISTS `getNthHighestSalary`;
-- Do not change delimter unless pasting into mysql terminal
CREATE FUNCTION `getNthHighestSalary`(`n` INT) RETURNS INT
BEGIN
  DECLARE x INT;
  SET x = n - 1;
  RETURN (SELECT DISTINCT `Salary` FROM `Employee` ORDER BY `Salary` DESC LIMIT 1 OFFSET x);
END;
-- Select highest salary
SELECT getNthHighestSalary(1) AS `HighestSalary`;
Table 1: 1 records

300 
-- Select second highest salary
SELECT getNthHighestSalary(2) AS `SecondHighestSalary`;
Table 2: 1 records

-- Select third highest salary
SELECT getNthHighestSalary(3) AS `ThirdHighestSalary`;
Table 3: 1 records
ThirdHighestSalary

-- Select fourth highest salary; returns NULL
SELECT getNthHighestSalary(4) AS `FourthHighestSalary`;
Table 4: 1 records
FourthHighestSalary
NA 
DROP TABLE `Employee`;
DROP FUNCTION `getNthHighestSalary`;
